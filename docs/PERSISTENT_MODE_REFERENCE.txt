📊 MONITORING

  sudo journalctl -u vega-server -f
  
  # Last 100 lines
  sudo journalctl -u vega-server -n 100
  
  # Last hour
  sudo journalctl -u vega-server --since "1 hour ago"
  
  # Memory stats
  curl http://localhost:8000/metrics | jq '.memory_manager'
  
  # Database size
  du -h /home/ncacord/Vega2.0/vega.db*

🧠 MEMORY MANAGEMENT
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  Thresholds:
    80%  = Warning  (gentle cleanup - compress old conversations)
    90%  = Critical (aggressive cleanup - compress + archive + vacuum)
  
  Context Lifecycle:
    Recent 50      → Full context (fast access)
    Older 50-100   → Compressed (prompts + summaries)
    Ancient 100+   → Archived as memory facts

💾 DATABASE OPERATIONS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  # Backup
  cp /home/ncacord/Vega2.0/vega.db ~/vega-backup-$(date +%Y%m%d).db
  
  # View conversations
  sqlite3 /home/ncacord/Vega2.0/vega.db "SELECT COUNT(*) FROM conversations;"
  
  # View by session
  sqlite3 /home/ncacord/Vega2.0/vega.db \
    "SELECT session_id, COUNT(*) FROM conversations GROUP BY session_id;"
  
  # Manual vacuum
  sqlite3 /home/ncacord/Vega2.0/vega.db "VACUUM;"

🔧 TROUBLESHOOTING
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  Service won't start:
    sudo journalctl -u vega-server -n 50
    sudo netstat -tlnp | grep 8000  # Check if port in use
  
  Out of memory:
    free -h  # Check available RAM
    # Stop Vega, clean DB, restart
    sudo systemctl stop vega-server
    sqlite3 /home/ncacord/Vega2.0/vega.db \
      "DELETE FROM conversations WHERE id < (SELECT MAX(id) - 50 FROM conversations);"
    sqlite3 /home/ncacord/Vega2.0/vega.db "VACUUM;"
    sudo systemctl start vega-server
  
  Web UI offline:
    curl http://localhost:8000/healthz  # Check if Vega responding
    sudo systemctl restart vega-server vega-dashboard
